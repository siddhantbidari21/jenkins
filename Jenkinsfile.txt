pipeline {
    agent {
        label 'Slave'
    }
    
    stages {
        stage('Checkout') {
            steps {
               checkout scm
            }
        }
        
        stage('Build') {
            steps {
               sh 'mvn clean compile'
            }
        }
        
        stage('Test') {
            steps {
               sh 'mvn test'
            }
        }
        
        stage('Package') {
            steps {
                 sh 'mvn package'
            }
        }
        
        stage('Deploy') {
            steps {
                sh 'cp target/project-1.0-SNAPSHOT.war /var/lib/tomcat9/webapps/'
            }
        }
    }
    
    post {
        success {
            echo 'Build and deployment successful.'
        }
        failure {
            echo 'Build or deployment failed.'
        }
    }
}
